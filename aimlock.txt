--== Aimlock Script ==--
local player = game.Players.LocalPlayer
local mouse = player:GetMouse()
local Aiming = false

function AimLock()
	local target
	local lastMagnitude = math.huge
	for _, v in pairs(game.Players:GetPlayers()) do
		if v ~= player and v.Character and v.Character:FindFirstChild("Head") then
			local charPos = v.Character.Head.Position
			local mousePos = mouse.Hit.p
			if (charPos - mousePos).Magnitude < lastMagnitude then
				lastMagnitude = (charPos - mousePos).Magnitude
				target = v
			end
		end
	end

	if target and target.Character and target.Character:FindFirstChild("Head") then
		local charPos = target.Character.Head.Position
		local cam = workspace.CurrentCamera
		local pos = cam.CFrame.Position
		workspace.CurrentCamera.CFrame = CFrame.new(pos, charPos)
	end
end

--== Input Handling ==--
local UserInputService = game:GetService("UserInputService")

UserInputService.InputBegan:Connect(function(input, gameProcessed)
	if not gameProcessed and input.KeyCode == Enum.KeyCode.E then
		Aiming = not Aiming
	end
end)

--== RenderStepped Loop (AimLock + Velocity Control) ==--
local Y = 0 -- Y-velocity override (prevents jumping/falling)

game:GetService("RunService").RenderStepped:Connect(function()
	-- Aimlock if enabled
	if Aiming then
		AimLock()
	end

	-- Velocity manipulation for all players except local
	for _, v in pairs(game.Players:GetPlayers()) do
		if v ~= player and v.Character and v.Character:FindFirstChild("HumanoidRootPart") then
			local old_vel = v.Character.HumanoidRootPart.Velocity
			v.Character.HumanoidRootPart.Velocity = Vector3.new(old_vel.X, Y, old_vel.Z)
		end
	end
end)
